// Example program
#include <iostream>
#include <time.h>


int main()
{
	srand(time(NULL));//для "настоящего" рандома при каждом запуске программы (нужно подключить time.h)

	int m[10];
	int* ar=m;

	for (int i = 0; i < sizeof(m)/sizeof(int); i++)
	{
		*(ar + i) = rand()%100;
		std::cout << *(ar + i) << ' ';
	}
	std::cout << std::endl;

	int max;
	max = *ar;
	int max2 = -2147483648;

	for (int i = 1; i < sizeof(m)/sizeof(int); i++)
	{
		
		if (*(ar + i) > max) {
			max2 = max;
			max = *(ar + i);
		}
		if (*(ar + i) < max && *(ar + i) >= max2) {
			max2 = *(ar + i);
		}
	}

	std::cout << "max=" << max << " max2=" << max2 << std::endl;

}
