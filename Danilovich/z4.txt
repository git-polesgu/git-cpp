#include <iostream>
#include <time.h>


int main()
{
	srand(time(NULL));//для "настоящего" рандома при каждом запуске программы (нужно подключить time.h)

	int m[9];
	int* ar = m;

	for (int i = 0; i < sizeof(m) / sizeof(int); i++)
	{
		*(ar + i) = rand() % 100;
		std::cout << *(ar + i) << ' ';
	}
	std::cout << std::endl;

	for (int i = 0; i < sizeof(m) / ( sizeof(int)*2); i++)
	{
		int temp = *(ar + i);
		*(ar + i) = *(ar + sizeof(m) / sizeof(int) - (i + 1));
		*(ar + sizeof(m) / sizeof(int) - (i + 1)) = temp;
	}

	for (int i = 0; i < sizeof(m) / sizeof(int); i++)
	{
		std::cout << *(ar + i) << ' ';
	}
}
