//я добавил '.' между римскими цифрами, чтобы при выводе на консоль буквы не слипались

#include <iostream>
#include <string>


int main()
{
	int number;
	std::cout << "number (0-3999) = ";
	std::cin >> number;

	std::string result="";

	if (number >= 1000) {
		int k = number / 1000;

		for (int i = 0; i < k; i++)
		{
			result += "M.";
			number -= 1000;
		}
	}
	if (number >= 100) {
		int k = number / 100;
		number -= 100 * k;

		switch (k)
		{
		case 1:
		case 2:
		case 3:
			for (int i = 0; i < k; i++)
			{
				result += "C.";
			}
			break;
		case 4:
			result += "C.D.";
			break;
		case 5:
		case 6:
		case 7:
		case 8:
			result += "D.";
			for (int i = 0; i < k-5; i++)
			{
				result += "C.";
			}
			break;
		case 9:
			result += "C.M.";
			break;
		}
	}
	if (number >= 10) {
		int k = number / 10;
		number -= 10 * k;


		switch (k)
		{
		case 1:
		case 2:
		case 3:
			for (int i = 0; i < k; i++)
			{
				result += "X.";
			}
			break;
		case 4:
			result += "X.L.";
			break;
		case 5:
		case 6:
		case 7:
		case 8:
			result += "L.";
			for (int i = 0; i < k - 5; i++)
			{
				result += "X.";
			}
			break;
		case 9:
			result += "X.C.";
			break;
		}

	}
	if (number >= 0) {
		switch (number)
		{
		case 1:
		case 2:
		case 3:
			for (int i = 0; i < number; i++)
			{
				result += "I.";
			}
			break;
		case 4:
			result += "I.V.";
			break;
		case 5:
		case 6:
		case 7:
		case 8:
			result += "V.";
			for (int i = 0; i < number - 5; i++)
			{
				result += "I.";
			}
			break;
		case 9:
			result += "I.X.";
			break;
		}

	}

	std::cout << result;
	return 0;
}
