#include <iostream>

using namespace std;

int main()
{
    int x;
    std::cout << "Введите число:\n";
    std::cin >> x;
    int y = x % 10;
    if (y == 0 || (y >= 5 && y <= 9) || (x >= 10 && x <=20))
    {
        std::cout
        << "Именительный: " << x << " деревьев \n"
        << "Родительный: " << x << " деревьев \n"
        << "Дательный: " << x << " деревьям \n"
        << "Винительный: " << x << " деревьев \n"
        << "Творительный: " << x << " деревьями \n"
        << "Предложный: " << x << " деревьях \n";
    }
    else if (y >= 2 && y <= 4)
    {
        std::cout
        << "Именительный: " << x << " дерева \n"
        << "Родительный: " << x << " деревьев \n"
        << "Дательный: " << x << " деревьям \n"
        << "Винительный: " << x << " дерева \n"
        << "Творительный: " << x << " деревьями \n"
        << "Предложный: " << x << " деревьях \n";
    }
    else
    {
        std::cout 
        << "Именительный: " << x << " дерево \n"
        << "Родительный: " << x << " дерева \n"
        << "Дательный: " << x << " дереву \n"
        << "Винительный: " << x << " дерево \n"
        << "Творительный: " << x << " деревом \n"
        << "Предложный: " << x << " дереве \n";        
    }
    return 0;
}
