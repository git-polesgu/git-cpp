#include <iostream>

using namespace std;

int main()
{
    double x;
    double y;
    char symbol;
    cout << "Введите первое число: \n";
    cin >> x;
    cout << "Введите второе число: \n";
    cin >> y;
    cout << "Введите символ арифметической операции: \n";
    cin >> symbol;
    
    switch (symbol)
    {
        case '+':
        cout << "Сумма: " << x + y;
        break;
        case '-':
        cout << "Разность: " << x - y;
        break;
        case '*':
        cout << "Произведение: " << x * y;
        break;
        case '/':
        cout << "Частное: " << x / y;
        break;
        default:
        cout << "Нет такой арифметической операции.";
        break;
    }

    return 0;
}
