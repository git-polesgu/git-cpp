// 2.5.cpp: определяет точку входа для консольного приложения.
//

#include "stdafx.h"
#include <iostream>


int _tmain(int argc, _TCHAR* argv[])
{
	int c;
	int n=5;
	int* arr=new int[n];
	for(int i=0;i<n;i++)
	{
		*(arr +i)=rand()%100;
		std::cout<<*(arr+i)<<" ";
	}
	std::cout<<""<<std::endl;
	for(int i=0;i<n;i++)
	{
		int min=i;
		int temp=*(arr+i);
		for(int j=i+1;j<n;++j)
		{
			if(*(arr+j)<temp)
			{
				min=j;
				temp=*(arr+j);
			}
		}
		*(arr+min)=*(arr+i);
		*(arr+i)=temp;
		std::cout<<*(arr+i)<<" ";
	}
	std::cin>>c;
	return 0;
}
